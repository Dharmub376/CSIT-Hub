/* Read a string from user and count the total length of string ,number of vowels(a,e,i,o,u), spaces , integers (0-9) from it*/
#include <stdio.h>
int main()
{
    char string[100];
    int i, vowel = 0, integers = 0, length = 0, spaces = 0;
    printf("Enter a string : ");
    scanf(" %[^\n]", string);
    for (i = 0; string[i] != '\0'; i++)
    {
        if (string[i] == 'a' || string[i] == 'e' || string[i] == 'i' || string[i] == 'o' || string[i] == 'u' || string[i] == 'A' || string[i] == 'E' || string[i] == 'I' || string[i] == 'O' || string[i] == 'U')
            vowel++;
        else if (string[i] >= '0' && string[i] <= '9')
            integers++;
        else if (string[i] == ' ')
            spaces++;

        length++;
    }
    printf("Total characters = %d.\nTotal integers = %d.\nTotal vowels = %d.\nTotal white spaces = %d.", length, integers, vowel, spaces);
    return 0;
}

#include <stdio.h>
double calculate(double a, char o, double b)
{
    switch (o)
    {
    case '+':
        return (a + b);
        break;
    case '-':
        return (a - b);
        break;
    case '*':
        return (a * b);
        break;
    case '/':
        if (b == 0)
        {
            printf("Cannot divide by 0 !!");
            return 0;
        }
        else
            return (a / b);
        break;
    default:
        printf("Invalid operator!!!");
        return 0;
        break;
    }
}
